using System;

public class School
{
    private int schoolID;
    private string schoolName;
    private string schoolNickname;
    private string schoolAddress;

	public School(int schoolID, string schoolName, string schoolNickname, string schoolAddress)
	{
        setSchoolID(schoolID);
        setSchoolName(schoolName);
        setSchoolNickname(schoolNickname);
        setSchoolAddress(schoolAddress);
	}


    //SETTERS
    protected void setSchoolID(schoolID)
    {
        this.schoolID = schoolID;
    }
    protected void setSchoolName(schoolName)
    {
        this.schoolName = schoolName;
    }
    protected void setSchoolNickname(schoolNickname)
    {
        this.schoolNickname = schoolNickname;
    }
    protected void setSchoolAddress(schoolAddress)
    {
        this.schoolAddress = schoolAddress;
    }
    
    //GETTERS
    protected int getSchoolID()
    {
        return this.schoolID;
    }
    protected string getSchoolName()
    {
        return this.schoolName;
    }
    protected string getSchoolNickname()
    {
        return this.schoolNickname;
    }
    protected string getSchoolAddress()
    {
        return this.schoolAddress;
    }
}
